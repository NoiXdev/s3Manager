# i18n (de/fr/pl/nl/ro) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make the renderer UI fully translatable with `react-i18next` and ship machine-translated baselines for German, French, Polish, Dutch, and Romanian, with English as the source and an OS-detected default overridable in Settings.

**Architecture:** Bundle all six locale JSON catalogs into a synchronously-initialized i18next instance (`fallbackLng: 'en'`). A `language` preference is added to `AppSettings` (mirroring `theme`) and applied via a `useLanguage` hook that resolves `'system'` from `navigator.language`. Every renderer string becomes a `t('key')` call; the English catalog reproduces today's exact strings so existing English-text test assertions keep passing.

**Tech Stack:** Electron, React 19, TypeScript, react-i18next + i18next, Tailwind, Vitest + Testing Library.

**Spec:** `docs/superpowers/specs/2026-06-11-i18n-design.md`

## Conventions (read before any task)

**Key naming:** single `translation` namespace, nested by feature area. Top-level groups:
`common`, `nav`, `settings`, `files`, `cors`, `objectLock`, `sync`, `accounts`, `connections`, `dashboard`, `buckets`, `transfer`.

`common.*` holds strings shared across features (e.g. `common.save`, `common.cancel`, `common.loading`, `common.close`, `common.settingsSaved`).

**Migration pattern for a component** (apply identically everywhere):
```tsx
import { useTranslation } from 'react-i18next';
// inside the component:
const { t } = useTranslation();
// replace:  <button>Cancel</button>
// with:     <button>{t('common.cancel')}</button>
```
For modules without a component (e.g. an array of label constants like `SectionNav`'s
`PRIMARY`), move the literal into the render where `t` is in scope, keying off a stable `id`.

**Interpolation / plurals:** embed values with `{{name}}`; use `_one`/`_other` keys for counts.
Example: `t('settings.showLicenses', { count: LICENSES.length })` with
`"showLicenses_one": "Show open source licenses ({{count}})"` and a matching `_other`.

**English values must equal the current on-screen string exactly** (including punctuation,
ellipses `…`, and casing) so existing assertions like `getByText('Files')` keep passing.

**Domain terms kept verbatim in all locales:** bucket, CORS, S3, ACL, presign, endpoint, sync.

import type { SettingsRepo } from '../storage/settingsRepo';

export type ThemePreference = 'system' | 'light' | 'dark';
export type LanguagePreference = 'system' | 'en' | 'de' | 'fr' | 'pl' | 'nl' | 'ro';

export interface AppSettings {
  presignExpirySeconds: number;
  theme: ThemePreference;
  language: LanguagePreference;
}
export interface AppInfo {
  version: string;
  encryptionAvailable: boolean;
  accountCount: number;
}

const DEFAULT_EXPIRY = 3600;
const MAX_EXPIRY = 604800; // S3's 7-day presign cap
const THEMES: ThemePreference[] = ['system', 'light', 'dark'];
const LANGUAGES: LanguagePreference[] = ['system', 'en', 'de', 'fr', 'pl', 'nl', 'ro'];

function isTheme(v: unknown): v is ThemePreference {
  return typeof v === 'string' && (THEMES as string[]).includes(v);
}

function isLanguage(v: unknown): v is LanguagePreference {
  return typeof v === 'string' && (LANGUAGES as string[]).includes(v);
}

export function readSettings(repo: SettingsRepo): AppSettings {
  const raw = repo.get('presignExpirySeconds');
  const n = raw !== undefined ? Number(raw) : NaN;
  const presignExpirySeconds = Number.isFinite(n) && n >= 1 && n <= MAX_EXPIRY ? n : DEFAULT_EXPIRY;
  const storedTheme = repo.get('theme');
  const theme: ThemePreference = isTheme(storedTheme) ? storedTheme : 'system';
  const storedLanguage = repo.get('language');
  const language: LanguagePreference = isLanguage(storedLanguage) ? storedLanguage : 'system';
  return { presignExpirySeconds, theme, language };
}

export function writeSettings(repo: SettingsRepo, patch: Partial<AppSettings>): AppSettings {
  if (patch.presignExpirySeconds !== undefined) {
    const clamped = Math.min(MAX_EXPIRY, Math.max(1, Math.round(patch.presignExpirySeconds)));
    repo.set('presignExpirySeconds', String(clamped));
  }
  if (patch.theme !== undefined && isTheme(patch.theme)) {
    repo.set('theme', patch.theme);
  }
  if (patch.language !== undefined && isLanguage(patch.language)) {
    repo.set('language', patch.language);
  }
  return readSettings(repo);
}
```

import { useEffect } from 'react';
import i18n from '../i18n';
import { SUPPORTED_LOCALES, type SupportedLocale } from '../i18n';
import type { LanguagePreference } from '../../main/settings/appSettings';

function resolveSystemLocale(): SupportedLocale {
  const primary = (navigator.language || 'en').split('-')[0].toLowerCase();
  return (SUPPORTED_LOCALES as readonly string[]).includes(primary)
    ? (primary as SupportedLocale)
    : 'en';
}

/**
 * Applies the language preference to i18next. When the preference is 'system'
 * (or undefined), it resolves the nearest supported locale from the OS via
 * navigator.language, falling back to English.
 */
export function useLanguage(preference: LanguagePreference | undefined): void {
  useEffect(() => {
    const pref = preference ?? 'system';
    const target = pref === 'system' ? resolveSystemLocale() : pref;
    if (i18n.language !== target) void i18n.changeLanguage(target);
  }, [preference]);
}
```

export type Section =
  | 'files' | 'dashboard' | 'objectLock' | 'cors' | 'sync' | 'settings' | 'connections';

const PRIMARY: { id: Section; key: string }[] = [
  { id: 'files', key: 'nav.files' },

      </button>

      <div role="separator" className="my-1 border-t border-slate-200 dark:border-slate-700" />
      {SECONDARY.map(renderItem)}
